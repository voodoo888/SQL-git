CREATE TRIGGER products_in BEFORE INSERT ON products
FOR EACH ROW 
BEGIN 
	IF NEW.name IS NULL AND NEW.desription IS NULL THEN 
	SIGNAL SQLSTATE '45000' SET message_text = 'Not null for name and desriptions';
	END IF;
END//
